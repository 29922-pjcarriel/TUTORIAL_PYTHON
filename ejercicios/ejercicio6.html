def validar_cedula_ec(cedula: str) -> bool:
    """
    Valida c√©dula ecuatoriana (10 d√≠gitos) con algoritmo m√≥dulo 10:
    - Dobla posiciones 0,2,4,6,8
    - Si el doble > 9, resta 9
    - Suma todo y calcula d√≠gito verificador
    """
    cedula = (cedula or "").strip()

    if len(cedula) != 10 or not cedula.isdigit():
        return False

    suma = 0
    for i in range(9):
        d = int(cedula[i])
        if i % 2 == 0:  # 0,2,4,6,8
            aux = d * 2
            if aux > 9:
                aux -= 9
            suma += aux
        else:
            suma += d

    decena = ((suma + 9) // 10) * 10
    resta = decena - suma
    if resta == 10:
        resta = 0

    return resta == int(cedula[9])

@app.route("/", methods=["GET", "POST"])
