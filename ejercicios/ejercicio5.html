Archivo: app.py
from flask import Flask, render_template, request

app = Flask(__name__)

def es_primo(n: int) -> bool:
    if n <= 1:
        return False
    for i in range(2, int(n ** 0.5) + 1):
        if n % i == 0:
            return False
    return True

# Acepta / y cualquier subruta
@app.route("/", methods=["GET", "POST"])
@app.route("/<path:subruta>", methods=["GET", "POST"])
def index(subruta=None):
    if request.method == "POST":
        try:
            numero = int(request.form.get("ingresoNumero", 0))
            primo = es_primo(numero)
            return render_template(
                "resultado.html",
                numero=numero,
                primo=primo
            )
        except Exception:
            return render_template(
                "resultado.html",
                numero=None,
                primo=None
            )

    return render_template("index.html")

if __name__ == "__main__":
    app.run(debug=True)

Este archivo recibe el número desde el formulario, verifica si es primo y envía el resultado a la vista.
